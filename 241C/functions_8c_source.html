399 {
400  //Reset gyro
401  SensorValue[gyro]=0;
402  //gyro takes degrees from 0-3600, so we multiply by 10 to get a gyro processable number
403  degree=degree*10;
404  //We want to slow down when we approach the target, so we calculate a first turn segment as 60% of the total
405  float first=degree*.6;
406  while(abs(SensorValue[gyro]) < first){ //Turn the first 60%
407  //Since it's turn right, we want to set right motors backwards and left motors forward.
408  frontLeftVal = speed;
409  frontRightVal = -speed;
410  backLeftVal = speed;
411  backRightVal = -speed;
412  }
413  while(abs(SensorValue[gyro]) <degree){ //Turn the remainin amount.
414  //We don't want the motors to run too slow, so we set a a safety net. The motor can't have a power less than 40.
415  if(speed*.35<40)//If 35% of the motor power is less than 40, set the power to 40.
416  {
417  frontLeftVal = 40;
