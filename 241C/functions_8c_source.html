234  // Drive motor until encoder has moved a number counts or
235  // timeout_in_seconds seconds have passed
236 
237  // Zero the encoder

241  // Run the motor forwards or backwards

251  // check encoder

253 
254  if((SensorValue[encoderL] + SensorValue[encoderR])/2 >= encoder_count){
255  break;

795  if(FW_half){

798  L1Error = L1SensorCurrentValue - FW_medSpeed;
799  L2Error = L2SensorCurrentValue - FW_medSpeed;

803  SensorValue[ledMed] = 1;

807  } else {

809  R2Error = R2SensorCurrentValue - FW_highSpeed;
810  L1Error = L1SensorCurrentValue - FW_highSpeed;
811  L2Error = L2SensorCurrentValue - FW_highSpeed;
812  if(((R1Error+R2Error+L1Error+L2Error)/4) <= FW_highSpeed+5){
813  SensorValue[ledHigh] = 1;
814  } else if(((R1Error+R2Error+L1Error+L2Error)/4) >= FW_highSpeed-5) {
815  SensorValue[ledHigh] = 1;
816  } else {
817  SensorValue[ledHigh] = 0;
818  }
819  }
820 
821  // integral - if Ki is not 0
822  if( pid_Ki != 0 ){
823 
824  // If we are inside controlable window then integrate the error
825  if( abs(R1Error) < PID_INTEGRAL_LIMIT ){
826  R1Integral = R1Integral + R1Error;
827  } else {
828  R1Integral = 0;
829  }

831  R2Integral = R2Integral + R2Error;
832  } else {

834  }

836  L1Integral = L1Integral + L1Error;
837  } else {
838  L1Integral = 0;
839  }

842  } else {
843  L2Integral = 0;
844  }
845  } else {
846  R1Integral = 0;
847  R2Integral = 0;
848  L1Integral = 0;
849  L2Integral = 0;
850  }

852  // calculate the derivative
853  R1Derivative = R1Error - R1LastError;
854  R1LastError = R1Error;
855 
856  R2Derivative = R2Error - R2LastError;
857  R2LastError = R2Error;
858

860  L1LastError = L1Error;
861

863  L2LastError = L2Error;

867  R2Drive = (pid_Kp * R2Error) + (pid_Ki * R2Integral) + (pid_Kd * R2Derivative);
868  L1Drive = (pid_Kp * L1Error) + (pid_Ki * L1Integral) + (pid_Kd * L1Derivative);
869  L2Drive = (pid_Kp * L2Error) + (pid_Ki * L2Integral) + (pid_Kd * L2Derivative);
