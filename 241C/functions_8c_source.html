411 void turnRightDegrees(float degree, float x=90)
412 {
413  //Reset gyro
414  SensorValue[gyro]=0;
415  //gyro takes degrees from 0-3600, so we multiply by 10 to get a gyro processable number
416  degree=degree*10;
417  //We want to slow down when we approach the target, so we calculate a first turn segment as 60% of the total
418  float first=degree*.6;
419  while(abs(SensorValue[gyro]) < first) //Turn the first 60%
420  {
421  //Since it's turn right, we want to set right motors backwards and left motors forward.
422  motor[FL] = x;
423  motor[FR] = -x;
424  motor[BL] = x;
425  motor[BR] = -x;
426  }
427  while(abs(SensorValue[gyro]) <degree) //Turn the remainin amount.
428  {
429  //We don't want the motors to run too slow, so we set a a safety net. The motor can't have a power less than 40.

431  {
432  motor[FL] = 40;

434  motor[BL] = 40;
