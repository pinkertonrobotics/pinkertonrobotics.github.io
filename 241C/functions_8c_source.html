403 void turnRightDegrees(float degree, float speed=90)
404 {
405  //Reset gyro
406  SensorValue[gyro]=0;
407  //gyro takes degrees from 0-3600, so we multiply by 10 to get a gyro processable number
408  degree=degree*10;
409  //We want to slow down when we approach the target, so we calculate a first turn segment as 60% of the total
410  float first=degree*.6;
411  while(abs(SensorValue[gyro]) < first){ //Turn the first 60%
412  //Since it's turn right, we want to set right motors backwards and left motors forward.
413  frontLeftVal = speed;
414  frontRightVal = -speed;
415  backLeftVal = speed;
416  backRightVal = -speed;
417  }
418  while(abs(SensorValue[gyro]) <degree){ //Turn the remainin amount.
419  //We don't want the motors to run too slow, so we set a a safety net. The motor can't have a power less than 40.
420  if(speed*.35<40)//If 35% of the motor power is less than 40, set the power to 40.
421  {
422  frontLeftVal = 40;
423  frontRightVal = -40;
424  backLeftVal = 40;
425  backRightVal = -40;
426  } else { //If not set it to 35%
427  frontLeftVal = speed*.35;
