415 void turnRightDegrees(float degree, float speed=90)
416 {
417  //Reset gyro
418  SensorValue[gyro]=0;
419  //gyro takes degrees from 0-3600, so we multiply by 10 to get a gyro processable number
420  degree=degree*10;
421  //We want to slow down when we approach the target, so we calculate a first turn segment as 60% of the total
422  float first=degree*.6;
423  while(abs(SensorValue[gyro]) < first){ //Turn the first 60%
424  //Since it's turn right, we want to set right motors backwards and left motors forward.
425  frontLeftVal = speed;
426  frontRightVal = -speed;
427  backLeftVal = speed;
428  backRightVal = -speed;
429  }
430  while(abs(SensorValue[gyro]) <degree){ //Turn the remainin amount.
431  //We don't want the motors to run too slow, so we set a a safety net. The motor can't have a power less than 40.
432  if(speed*.35<40)//If 35% of the motor power is less than 40, set the power to 40.
433  {
434  frontLeftVal = 40;
435  frontRightVal = -40;
436  backLeftVal = 40;
437  backRightVal = -40;
438  } else { //If not set it to 35%
439  frontLeftVal = speed*.35;

443  }
