733 task FW_pidController()
734 {
735 
736  // Init the variables - thanks Glenn :)
737  //pidLastError = 0;
738  //pidIntegral = 0;
739 
740  while( true ) {
741  // Is PID control active ?
742  swag += 1;
743  if( FW_running ) {
744  // Read the sensor value and scale
745  pidSensorCurrentValue = ((abs(SensorValue[flyR2IEM]) + abs(SensorValue[flyL2IEM])) / 2) * PID_SENSOR_SCALE;
746 
747  // since flywheel is continuously spinning, reset encoders
748  SensorValue[flyR2IEM] = 0;
749  SensorValue[flyL2IEM] = 0;
750 
751  // calculate error
752  if(FW_half){
753  pidError = pidSensorCurrentValue - FW_medSpeed;
754  if(pidError <= FW_medSpeed+5){
755  SensorValue[ledMed] = 1;
756  } else if(pidError >= FW_medSpeed-5) {
757  SensorValue[ledMed] = 1;
758  } else {
759  SensorValue[ledMed] = 0;
760  }
761  } else {
762  pidError = pidSensorCurrentValue - FW_highSpeed;
763  if(pidError <= FW_highSpeed+5){
764  SensorValue[ledHigh] = 1;
765  } else if(pidError >= FW_highSpeed-5) {
766  SensorValue[ledHigh] = 1;
767  } else {
768  SensorValue[ledHigh] = 0;
769  }
770  }
771 
772  // integral - if Ki is not 0
773  if( pid_Ki != 0 )
774  {
775  // If we are inside controlable window then integrate the error
776  if( abs(pidError) < PID_INTEGRAL_LIMIT )
777  pidIntegral = pidIntegral + pidError;
778  else
779  pidIntegral = 0;
780  }
781  else
782  pidIntegral = 0;
783 
784  // calculate the derivative
785  pidDerivative = pidError - pidLastError;
786  pidLastError = pidError;
787 
788  // calculate drive
789  pidDrive = (pid_Kp * pidError) + (pid_Ki * pidIntegral) + (pid_Kd * pidDerivative);
790 
791  // limit drive
792  if( pidDrive > PID_DRIVE_MAX )
793  pidDrive = PID_DRIVE_MAX;
794  if( pidDrive < PID_DRIVE_MIN )
795  pidDrive = PID_DRIVE_MIN;
796 
797  // send to flywheel
798  motor[flyR1] = pidDrive * PID_MOTOR_SCALE;
799  motor[flyR2] = pidDrive * PID_MOTOR_SCALE;
800  motor[flyL1] = pidDrive * PID_MOTOR_SCALE;
801  motor[flyL2] = pidDrive * PID_MOTOR_SCALE;
802  motor[topIntake] = (pidDrive * PID_MOTOR_SCALE) - ((pidDrive * PID_MOTOR_SCALE)/5);
803  } else {
804  // clear all
805  pidError = 0;
806  pidLastError = 0;
807  pidIntegral = 0;
808  pidDerivative = 0;
809  pidDrive = 0;
810  motor[flyR1] = 0;
811  motor[flyR2] = 0;
812  motor[flyL1] = 0;
813  motor[flyL2] = 0;
814  motor[topIntake] = 0;
815  }
816  // Run at 4Hz
817  wait1Msec( 250 );
818  }
819 }
