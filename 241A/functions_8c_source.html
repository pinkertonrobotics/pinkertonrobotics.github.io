358 void turnRightDegrees(float degree, float speed=90)
359 {
360  //Reset gyro
361  SensorValue[gyro]=0;
362  //gyro takes degrees from 0-3600, so we multiply by 10 to get a gyro processable number
363  degree=degree*10;
364  //We want to slow down when we approach the target, so we calculate a first turn segment as 60% of the total
365  float first=degree*.6;
366  while(abs(SensorValue[gyro]) < first){ //Turn the first 60%
367  //Since it's turn right, we want to set right motors backwards and left motors forward.
368  frontLeftVal = speed;
369  frontRightVal = -speed;
370  backLeftVal = speed;
371  backRightVal = -speed;
372  }
373  while(abs(SensorValue[gyro]) <degree){ //Turn the remainin amount.
374  //We don't want the motors to run too slow, so we set a a safety net. The motor can't have a power less than 40.
