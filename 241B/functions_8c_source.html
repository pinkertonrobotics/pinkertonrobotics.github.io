347 void turnRightDegrees(float degree, float speed=90)
348 {
349  //Reset gyro
350  SensorValue[gyro]=0;
351  //gyro takes degrees from 0-3600, so we multiply by 10 to get a gyro processable number
352  degree=degree*10;
353  //We want to slow down when we approach the target, so we calculate a first turn segment as 60% of the total
354  float first=degree*.6;
355 
356  while(abs(SensorValue[gyro]) < first){ //Turn the first 60%
357  //Since it's turn right, we want to set right motors backwards and left motors forward.
358  motor[RD] = speed;
359  motor[LD] = speed;
360  }
361  while(abs(SensorValue[gyro]) < degree){ //Turn the remainin amount.
362  //We don't want the motors to run too slow, so we set a a safety net. The motor can't have a power less than 40.
363  if(speed*.35<40)//If 35% of the motor power is less than 40, set the power to 40.
364  {
365  motor[RD] = 40;
366  motor[LD] = 40;
367  } else { //If not set it to 35%
368  motor[LD] = speed*.35;
369  motor[RD] = -speed*.35;
370  }
371  }
372  clearMotors();
373 }
