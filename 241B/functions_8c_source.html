352 void turnRightDegrees(float degree, float speed=90)
353 {
354  //Reset gyro
355  SensorValue[gyro]=0;
356  //gyro takes degrees from 0-3600, so we multiply by 10 to get a gyro processable number
357  degree=degree*10;
358  //We want to slow down when we approach the target, so we calculate a first turn segment as 60% of the total
359  float first=degree*.6;
360 
361  while(abs(SensorValue[gyro]) < first){ //Turn the first 60%
362  //Since it's turn right, we want to set right motors backwards and left motors forward.
363  motor[RD] = speed;
364  motor[LD] = speed;
365  }
366  while(abs(SensorValue[gyro]) < degree){ //Turn the remainin amount.
367  //We don't want the motors to run too slow, so we set a a safety net. The motor can't have a power less than 40.
368  if(speed*.35<40)//If 35% of the motor power is less than 40, set the power to 40.
369  {
370  motor[RD] = 40;
371  motor[LD] = 40;
372  } else { //If not set it to 35%
373  motor[LD] = speed*.35;
