348 void turnRightDegrees(float degree, float speed=90)
349 {
350  //Reset gyro
351  SensorValue[gyro]=0;
352  //gyro takes degrees from 0-3600, so we multiply by 10 to get a gyro processable number
353  degree=degree*10;
354  //We want to slow down when we approach the target, so we calculate a first turn segment as 60% of the total
355  float first=degree*.6;
356 
357  while(abs(SensorValue[gyro]) < first){ //Turn the first 60%
358  //Since it's turn right, we want to set right motors backwards and left motors forward.
359  motor[RD] = speed;
360  motor[LD] = speed;
361  }
362  while(abs(SensorValue[gyro]) < degree){ //Turn the remainin amount.
363  //We don't want the motors to run too slow, so we set a a safety net. The motor can't have a power less than 40.
364  if(speed*.35<40)//If 35% of the motor power is less than 40, set the power to 40.
365  {
366  motor[RD] = 40;
367  motor[LD] = 40;
368  } else { //If not set it to 35%
369  motor[LD] = speed*.35;
370  motor[RD] = -speed*.35;
371  }
372  }
373  clearMotors();
