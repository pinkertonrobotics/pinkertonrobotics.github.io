416 void turnRightDegrees(float degree, float x=90)
417 {
418  //Reset gyro
419  SensorValue[gyro]=0;
420  //gyro takes degrees from 0-3600, so we multiply by 10 to get a gyro processable number
421  degree=degree*10;
422  //We want to slow down when we approach the target, so we calculate a first turn segment as 60% of the total
423  float first=degree*.6;
424  while(abs(SensorValue[gyro]) < first) //Turn the first 60%
425  {
426  //Since it's turn right, we want to set right motors backwards and left motors forward.
427  motor[FL] = x;
428  motor[FR] = -x;
429  motor[BL] = x;
430  motor[BR] = -x;
431  }
432  while(abs(SensorValue[gyro]) <degree) //Turn the remainin amount.
433  {
434  //We don't want the motors to run too slow, so we set a a safety net. The motor can't have a power less than 40.
435  if(x*.35<40)//If 35% of the motor power is less than 40, set the power to 40.
436  {
437  motor[FL] = 40;
438  motor[FR] = -40;
439  motor[BL] = 40;
440  motor[BR] = -40;
441  }
442  else //If not set it to 35%
443  {
444  motor[FL] = x*.35;
445  motor[FR] = -x*.35;
446  motor[BL] = x*.35;
447  motor[BR] = -x*.35;
448  }
449  }
450  clearMotor();
451 }
