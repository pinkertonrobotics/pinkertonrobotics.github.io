387 void turnRightDegrees(float degree, float x=90)
388 {
389  //Reset gyro
390  SensorValue[gyro]=0;
391  //gyro takes degrees from 0-3600, so we multiply by 10 to get a gyro processable number
392  degree=degree*10 + 28;
393  //We want to slow down when we approach the target, so we calculate a first turn segment as 60% of the total
394  float first=degree*.35;
395  while(abs(SensorValue[gyro]) < first) //Turn the first 60%
396  {
397  //Since it's turn right, we want to set right motors backwards and left motors forward.
398  motor[LF] = x;
399  motor[RF] = -x;
400  motor[LB] = x;
401  motor[RB] = -x;
402  }
403  while(abs(SensorValue[gyro]) <degree) //Turn the remainin amount.
404  {
405  //We don't want the motors to run too slow, so we set a a safety net. The motor can't have a power less than 40.
406  if(x*.35<40)//If 35% of the motor power is less than 40, set the power to 40.
407  {
408  motor[LF] = 64;
409  motor[RF] = -64;
410  motor[LB] = 64;
411  motor[RB] = -64;
412  }
413  else //If not set it to 35%
414  {
415  motor[LF] = x*.35;
416  motor[RF] = -x*.35;
417  motor[LB] = x*.35;
418  motor[RB] = -x*.35;
419  }
420  }
421  clearMotor();
422 }
