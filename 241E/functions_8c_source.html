365 void turnRightDegrees(float degree, float speed=90)
366 {
367  //Reset gyro
368  SensorValue[gyro]=0;
369  //gyro takes degrees from 0-3600, so we multiply by 10 to get a gyro processable number
370  degree=degree*10;
371  //We want to slow down when we approach the target, so we calculate a first turn segment as 60% of the total
372  float first=degree*.6;
373  while(abs(SensorValue[gyro]) < first){ //Turn the first 60%
374  //Since it's turn right, we want to set right motors backwards and left motors forward.
375  frontLeftVal = speed;
376  frontRightVal = -speed;
377  backLeftVal = speed;
378  backRightVal = -speed;
379  }
380  while(abs(SensorValue[gyro]) <degree){ //Turn the remainin amount.
381  //We don't want the motors to run too slow, so we set a a safety net. The motor can't have a power less than 40.
382  if(speed*.35<40)//If 35% of the motor power is less than 40, set the power to 40.
383  {
384  frontLeftVal = 40;
385  frontRightVal = -40;
386  backLeftVal = 40;
387  backRightVal = -40;
388  } else { //If not set it to 35%
389  frontLeftVal = speed*.35;
390  frontRightVal = -speed*.35;
391  backLeftVal = speed*.35;
392  backRightVal = -speed*.35;
393  }
